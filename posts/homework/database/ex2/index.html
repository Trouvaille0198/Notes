    SELECT
	e.kh,
	c.km,
	t.xm 
FROM
	e
	INNER JOIN s ON e.xh = s.xh
	INNER JOIN t ON e.gh = t.gh
	INNER JOIN c ON e.kh = c.kh 
WHERE
	s.xb = '男' 
	AND t.xl = '教授' 
	AND zpcj >= 60 
ORDER BY
	t.gh ASC,
	c.kh DESC
