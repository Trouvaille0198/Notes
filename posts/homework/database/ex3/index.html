    SELECT
	s.xh,
	s.xm,
	avg( e.zpcj ) 
FROM
	s
	INNER JOIN e ON s.xh = e.xh 
WHERE
	( 2021-year ( s.csrq ) )< ( SELECT avg( 2021-year ( c.csrq ) ) FROM s AS c WHERE c.yxh = s.yxh ) 
	AND NOT EXISTS (
	SELECT
		* 
	FROM
		e AS a 
	WHERE
		a.xh = e.xh 
		AND a.zpcj <(
		SELECT
			avg( b.zpcj ) 
		FROM
			e AS b 
		WHERE
			b.kh = a.kh 
		) 
	) 
GROUP BY
	s.xh,
	s.xm 
ORDER BY
	(
	2021-year ( s.csrq ) 
	)

    SELECT
	e1.xq,
	e1.kh,
	e1.xh,
	e1.zpcj,	
	count(*) AS pm 
FROM
	e AS e1,
	e AS e2 
WHERE
	(
		e2.zpcj > e1.zpcj 
	OR ( e1.zpcj = e2.zpcj AND e1.xh = e2.xh )) 
	AND e1.kh = e2.kh 
	AND e1.xq = e2.xq 
GROUP BY
	e1.xq,
	e1.kh,
	e1.xh,
	e1.zpcj
ORDER BY
	e1.xq DESC,
	e1.kh ASC,
	e1.zpcj DESC;

    SELECT
	e.xq,
	s.xh,
	e.kh 
FROM
	e
	INNER JOIN s ON e.xh = s.xh
	INNER JOIN t ON e.gh = t.gh
	INNER JOIN c ON e.kh = c.kh 
WHERE
	s.xb = '男' 
	AND s.yxh = 1 
	AND t.xl = '教授' 
	AND t.yxh = 1 
	AND e.zpcj < 60 
ORDER BY
	e.xq,
	s.xh
