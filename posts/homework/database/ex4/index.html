    CREATE VIEW view1 ( xh, xm, xb, sjhm, km, zpcj ) AS (
	SELECT
		e.xh,
		xm,
		xb,
		sjhm,
		km,

	FROM
		e
		INNER JOIN s ON e.xh = s.xh
		INNER JOIN c ON e.kh = c.kh 
	WHERE
	e.zpcj < 60 
	AND s.yxh = 1)
