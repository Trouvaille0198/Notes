剑指 Offer 16. 数值的整数次方

easy

实现 pow(x, n) ，即计算 x 的 n 次幂函数（即，xn）。不得使用库函数，同时不需要考虑大数问题。

func myPow(x float64, n int) float64 {
   if x == 0 {
      return 0
   }
   if n < 0 {
      x, n = 1/x, -n
   }
   res := 1.0
   var b int // 二进制n的某一位 0或1
   for n != 0 {
      b = n & 1 // 获取二进制n最后一位
      if b != 0 {
         res *= x
      }
      n = n >> 1 // n右移一位
      x = x * x
   }
   return res
}
